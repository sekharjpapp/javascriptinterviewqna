3. Conditional Logic

```
// If-else statement
if (age >= 18) {
  console.log("You can vote!");
} else {
  console.log("Wait till you're 18");
}

// Switch statement
let day = "Monday";
switch(day) {
  case "Monday":
    console.log("Start of week");
    break;
  default:
    console.log("Other day");
}
```
